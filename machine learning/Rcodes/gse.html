kegg_go <- function(rTable) {
  rTable <- arrange(rTable, padj)
  info <- rTable
  GO_database <- 'org.Mm.eg.db' 
  KEGG_database <- 'mmu'
  dim(info)
  
  gene <- bitr(info$SYMBOL,
               fromType = 'SYMBOL',
               toType = 'ENTREZID',
               OrgDb = GO_database)
  
  names(info) <- c('SYMBOL', 'Log2FoldChange', 'pvalue', 'padj')
  info_merge <- merge(info, gene, by = 'SYMBOL')
  GSEA_input <- info_merge$Log2FoldChange
  names(GSEA_input) <- info_merge$ENTREZID
  GSEA_input <- sort(GSEA_input, decreasing = TRUE)
  GSEA_KEGG <- gseKEGG(GSEA_input, organism = KEGG_database, pvalueCutoff = 0.05)
  
  KEGG <- enrichKEGG(gene$ENTREZID,
                     organism = KEGG_database,
                     pvalueCutoff = 0.05,
                     qvalueCutoff = 0.05)
  GO <- enrichGO(gene$ENTREZID,
                OrgDb = GO_database,
                keyType = "ENTREZID",
                ont = "ALL",
                pvalueCutoff = 0.05,
                qvalueCutoff = 0.05,
                readable = T)
  
  p1<-barplot(GO, split="ONTOLOGY")+facet_grid(ONTOLOGY~., scale="free")#柱状图
  p1
  p2<-barplot(KEGG,showCategory = 30,title = 'KEGG Pathway')
  p2
  
  p3<-dotplot(GO, split="ONTOLOGY")+facet_grid(ONTOLOGY~., scale="free")#点状图
  p3
  p4<-dotplot(KEGG)
  p4
  
  p5<-enrichplot::cnetplot(KEGG,circular=TRUE,colorEdge = TRUE,edge_color = 'grey',
                           color_category = "#00BFFF",
                           color_gene = "#FF3030",
                           color.params = list(foldChange = NULL, edge = FALSE, category = "#00BFFF", gene =
                                                 "#B3B3B3"))
  p5
  p6<-enrichplot::cnetplot(GO,circular=TRUE,colorEdge = TRUE,edge_color = 'grey',
                           color_category = "#00BFFF",
                           color_gene = '#FF3030',
                           color.params = list(foldChange = NULL, edge = FALSE, category = "#00BFFF", gene =
                                                 "#B3B3B3"))
  p6 
  
  # Ridgeplot图
  p7 <- ridgeplot(GSEA_KEGG) +
    scale_fill_gradient(low = '#FF3030', high = '#00BFFF') + # 渐变填充颜色
    theme_minimal()
  
  p8 <- GSEA_KEGG
  
  return(list(p1, p2, p3, p4, p5, p6, p7, p8))
}
